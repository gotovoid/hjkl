2. 编写脚本

#!/usr/bin/env gnuplot

set terminal png
set output 'editor.png'
set title "What's your favorite text editor?" font 'Ubuntu,16'
set xtics scale 0 rotate by -45
set style fill solid 0.25 noborder
set box 0.5
set rmargin 5
set bmargin 5
set grid y

plot 'editor.csv' using 2:xtic(1) with boxes linecolor rgb 'blue' title 'Votes',\
     '' using 0:($2+50):3 with labels notitle

3. 代码解释

#!/usr/bin/env gnuplot

# 输出类型
set terminal png
# 输出文件名
set output 'editor.png'
# 标题及其字体
set title "What's your favorite text editor?" font 'Ubuntu,16'
# X轴刻度标签顺时针旋转45°, 并且隐藏刻度线
set xtics scale 0 rotate by -45
# 填充颜色, 并隐藏边框
set style fill solid 0.25 noborder
# 矩形框的宽度
set box 0.5
# 右边距
set rmargin 5
# 下边距
set bmargin 5
# 显示Y轴网格
set grid y

# 先绘制boxes, 再绘制labels
plot 'editor.csv' using 2:xtic(1) with boxes linecolor rgb 'blue' title 'Votes',\
     '' using 0:($2+50):3 with labels notitle

4. 运行脚本

chmod +x editor.plt
./editor.plt
display editor.png
