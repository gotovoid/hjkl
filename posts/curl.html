cURL cheat sheet 


curl http://www.site.com/

curl http://www.site.com/ -o index.html

curl -O http://www.site.com/logo.png

curl 'http://www.site.com/download/{cat,dog}/[001-999].png' -o '#1_#2.png'

xargs -a urls.txt -n1 -P0 curl -O

curl -H 'User-Agent: Mozilla/5.0' http://www.site.com/

curl -e http://www.google.com/ http://www.site.com/

curl -d 'q=hello' http://www.google.com/

curl -L http://www.old-site.com/

curl -I http://www.site.com/

curl -c cookie.txt -u user:passwd http://www.site.com/login

curl -b cookie.txt http://www.site.com/

curl -F 'file=@localfile.txt;type=plain/text' http://www.site.com/upload

curl --trace-ascii - http://www.site.com/ -o index.html
